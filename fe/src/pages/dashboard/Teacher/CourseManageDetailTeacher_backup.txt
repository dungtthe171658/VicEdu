import { useEffect, useState } from "react";
import { useNavigate, useParams } from "react-router-dom";
import courseTeacherApi from "../../../api/courseTeacherApi";
import type { Course } from "../../../types/course";
import CourseForm from "../../../components/courses/CourseForm";
import LessonManager from "../../../components/courses/LessonManager";

export default function CourseManageDetail() {
  const { courseId } = useParams();
  const navigate = useNavigate();
  const [course, setCourse] = useState<Course | null>(null);
  const [loading, setLoading] = useState(true);
  const [saving, setSaving] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const load = async () => {
    if (!courseId) return;
    setLoading(true);
    setError(null);
    try {
      const res = await courseTeacherApi.getById(courseId);
      setCourse(res.data ?? res); // support either shape
    } catch (e: any) {
      setError(e?.message || "KhÃ´ng táº£i Ä‘Æ°á»£c khÃ³a há»c");
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    load();
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [courseId]);

  const handleSave = async (data: Partial<Course>) => {
    if (!courseId) return;
    setSaving(true);
    setError(null);
    try {
      await courseTeacherApi.update(courseId, data);
      await load();
    } catch (e: any) {
      setError(e?.message || "LÆ°u khÃ³a há»c tháº¥t báº¡i");
    } finally {
      setSaving(false);
    }
  };

  const submitForPublish = async () => {
    if (!courseId) return;
    try {
      await courseTeacherApi.requestPublish(courseId);
      await load();
    } catch (e: any) {
      alert(e?.message || 'Gá»­i yÃªu cáº§u publish tháº¥t báº¡i');
    }
  };

  return (
    <div className="p-4">
      {(course as any)?.has_pending_changes && (
        <div style={{ background: '#fff7ed', border: '1px solid #fdba74', color: '#9a3412', padding: 8, borderRadius: 8, marginBottom: 12 }}>
          Thay Ä‘á»•i cá»§a báº¡n Ä‘ang chá» Admin phÃª duyá»‡t.
        </div>
      )}
      <div style={{ display: "flex", alignItems: "center", gap: 12, marginBottom: 16 }}>
        <button onClick={() => navigate("/teacher/manage-courses")}>
          â† Quay láº¡i
        </button>
        <h2 style={{ margin: 0 }}>Thiáº¿t káº¿ khÃ³a há»c</h2>
      </div>

      {loading ? (
        <div>Äang táº£i khÃ³a há»c...</div>
      ) : error ? (
        <div style={{ color: "red" }}>{error}</div>
      ) : !course ? (
        <div>KhÃ´ng tÃ¬m tháº¥y khÃ³a há»c.</div>
      ) : (
        <div style={{ display: "grid", gridTemplateColumns: "1.2fr 1fr", gap: 20 }}>
          {!((course as any)?.is_published) && (
            <div style={{ gridColumn: '1 / -1', background: '#fff', borderRadius: 12, padding: 12, display: 'flex', gap: 8, alignItems: 'center' }}>
              {(course as any)?.publish_requested_at ? (
                <span>ÄÃ£ gá»­i yÃªu cáº§u publish lÃºc {(course as any)?.publish_requested_at ? new Date((course as any).publish_requested_at).toLocaleString('vi-VN') : ''}</span>
              ) : (
                <button onClick={submitForPublish}>Gá»­i yÃªu cáº§u publish</button>
              )}
            </div>
          )}
          <div style={{ background: "#fff", borderRadius: 12, padding: 16 }}>
            <h3 style={{ marginTop: 0 }}>ThÃ´ng tin khÃ³a há»c</h3>
            <CourseForm\n              initialData={{\n                ...course,\n                category_id: (() => {\n                  const cat = (course as any)?.category;\n                  if (Array.isArray(cat) && cat.length > 0) return String(cat[0]?._id || cat[0]);\n                  if ((course as any)?.category_id) return String((course as any).category_id);\n                  return "";\n                })(),\n              }}\n              onSubmit={handleSave}\n            />
            {saving && <div style={{ marginTop: 8 }}>Äang lÆ°u...</div>}
          </div>
          {(course as any)?.has_pending_changes && (course as any)?.draft && (
            <div style={{ background: '#fff', borderRadius: 12, padding: 16 }}>
              <h3 style={{ marginTop: 0 }}>Báº£n nhÃ¡p Ä‘ang chá» duyá»‡t</h3>
              <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr 1fr', gap: 8 }}>
                <div style={{ fontWeight: 600 }}>Field</div>
                <div style={{ fontWeight: 600 }}>Before</div>
                <div style={{ fontWeight: 600 }}>After (draft)</div>
                {Object.entries((course as any).draft).map(([k, v]) => (
                  <>
                    <div key={k + '-f'} style={{ fontFamily: 'monospace' }}>{k}</div>
                    <div key={k + '-b'} style={{ whiteSpace: 'pre-wrap' }}>{String(((course as any)[k] !== undefined ? (course as any)[k] : ''))}</div>
                    <div key={k + '-a'} style={{ whiteSpace: 'pre-wrap' }}>{String(v as any)}</div>
                  </>
                ))}
              </div>
              <div style={{ marginTop: 8, color: '#6b7280', fontSize: 12 }}>
                Gá»­i lÃºc: {((course as any).pending_at ? new Date((course as any).pending_at).toLocaleString() : '')}
              </div>
            </div>
          )}
          <div style={{ background: "#fff", borderRadius: 12, padding: 16 }}>
            <LessonManager courseId={course._id} />
          </div>
        </div>
      )}
    </div>
  );
}
